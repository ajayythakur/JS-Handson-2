Q1. Write a program to make a simple calculator using switch statement that takes operator and two numbers from user and print the result after operation.

Ans. 

        const Calculator = (A, B, C) => 
        {
            switch(A){
              case"+":
                var add=B+C;
                return(add);
                break;
                case"*":
                  var multiply=B*C;
                  return(multiply);
                  break;
                  case"-":
                    var diff=B-C;
                    return(diff);
                    break;
                    case"/":
                      var div=B/C;
                      return(div);
                      break;
                      default:
                      var def="Enter Operator Number1 Number2";
                      return(def);
                      break;        
            } 
        };
